		\vspace{10pt}
		\begin{center}

			\left[\begin{array}{ccc}
				l_1*cos(\alpha_1)+l_2*cos(\alpha_1+\alpha_2)
				& l_2*cos(\alpha_1+\alpha_2) & 0 \\
				
				-l_1*sin(\alpha_1)-l_2*sin(\alpha_1+\alpha_2)
				& -l_2*sin(\alpha_1+\alpha_2) & 0 \\
				
				1 & 1 & 1
			\end{array}\right]$
		\end{center}
